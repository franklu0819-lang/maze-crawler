## 一、核心认知：这个游戏的本质

游戏胜负由三个要素决定：**Factory 是否存活 > 总能量 > 单位数**。所以所有决策都应该服务于这个优先级树，而不是为了战斗而战斗。

## 二、地图与信息系统（基础设施）

这是最容易被新手忽略但最关键的部分。

**建立全局地图对象**，在每一帧更新：

```python
class GameState:
    def __init__(self, config):
        self.width = config.width
        self.known_walls = {}       # (col, row) → bitfield，永久记忆
        self.known_mines = {}       # (col, row) → [energy, max, owner]
        self.enemy_last_seen = {}   # uid → (col, row, turn)
        self.step = 0

    def update(self, obs):
        self.step += 1
        self.south = obs.southBound
        self.north = obs.northBound
        
        # 合并墙体信息（永久记忆）
        for i, w in enumerate(obs.walls):
            if w != -1:
                row = self.south + i // self.width
                col = i % self.width
                self.known_walls[(col, row)] = w
        
        # 更新矿场信息
        for key, data in obs.mines.items():
            col, row = map(int, key.split(','))
            self.known_mines[(col, row)] = data
        
        # 追踪敌方机器人
        for uid, data in obs.robots.items():
            if data[4] != obs.player:
                self.enemy_last_seen[uid] = (data[1], data[2], self.step)
```

**BFS 寻路**（考虑墙体）：

```python
from collections import deque

def bfs(state, start, goal, avoid_cells=set()):
    """返回从 start 到 goal 的第一步方向字符串"""
    directions = [("NORTH", 0, 1), ("SOUTH", 0, -1), 
                  ("EAST", 1, 0), ("WEST", -1, 0)]
    wall_bits  = {"NORTH": 1, "EAST": 2, "SOUTH": 4, "WEST": 8}
    
    queue = deque([(start, [])])
    visited = {start}
    
    while queue:
        (col, row), path = queue.popleft()
        if (col, row) == goal:
            return path[0] if path else "IDLE"
        
        w = state.known_walls.get((col, row), 0)
        for name, dc, dr in directions:
            if w & wall_bits[name]:
                continue  # 有墙
            nc, nr = col + dc, row + dr
            if (nc, nr) in visited or (nc, nr) in avoid_cells:
                continue
            if not (state.south <= nr <= state.north):
                continue
            if not (0 <= nc < state.width):
                continue
            visited.add((nc, nr))
            queue.append(((nc, nr), path + [name]))
    
    return "IDLE"  # 不可达
```

## 三、Factory 的决策树

Factory 是你唯一不能丢的单位。其决策按严格优先级排列：

```python
def factory_action(uid, data, state, obs, config):
    col, row, energy = data[1], data[2], data[3]
    jump_cd, build_cd = data[6], data[7]
    
    # ① 危险区检测：距南边界 <= 3 行，必须逃
    danger_rows = 3
    if row - state.south <= danger_rows:
        w = state.known_walls.get((col, row), 0)
        if not (w & 1) and jump_cd == 0:
            # 跳跃比走路快
            return "JUMP_NORTH"
        elif not (w & 1):
            return "NORTH"
        elif jump_cd == 0:
            return "JUMP_NORTH"   # 穿墙跳跃
        else:
            # 寻找最近出路
            return find_escape(col, row, state)
    
    # ② 建造时机：能量充足且北方无墙
    w = state.known_walls.get((col, row), 0)
    north_clear = not (w & 1)
    
    if build_cd == 0 and north_clear:
        # 建造优先级：Scout（视野）→ Miner（有矿点）→ Worker（开路）
        if has_nearby_mining_node(col, row, obs, state) and energy >= config.minerCost:
            return "BUILD_MINER"
        elif scout_count(obs) < 3 and energy >= config.scoutCost:
            return "BUILD_SCOUT"
        elif energy >= config.workerCost:
            return "BUILD_WORKER"
    
    # ③ 正常北移
    if north_clear:
        return "NORTH"
    elif jump_cd == 0:
        return "JUMP_NORTH"
    else:
        return "IDLE"
```

**关键点：** `JUMP_NORTH` 会穿墙，但如果落点在 northBound 之外会直接摧毁 Factory，**必须检查 row + 2 <= northBound**。

## 四、经济引擎：Scout + Miner 协同

这是高分的核心。一个运作良好的矿场每回合产出 50 能量，远超 Scout 随机拾取晶石。

**Scout 行为：优先探索，次要采晶石**

```python
def scout_action(uid, data, state, obs, config):
    col, row = data[1], data[2]
    
    # 目标优先级：矿点 > 晶石 > 未探索区域 > 北移
    
    # 找最近可见矿点
    if obs.miningNodes:
        nearest_node = min(obs.miningNodes.keys(), 
            key=lambda k: manhattan(k, col, row))
        nc, nr = map(int, nearest_node.split(','))
        return bfs(state, (col, row), (nc, nr))
    
    # 找最近晶石
    if obs.crystals:
        nearest_crystal = min(obs.crystals.keys(),
            key=lambda k: manhattan(k, col, row))
        cc, cr = map(int, nearest_crystal.split(','))
        return bfs(state, (col, row), (cc, cr))
    
    # 向未探索的北方移动，覆盖视野盲区
    return explore_north(col, row, state, obs)
```

**Miner 行为：直奔矿点，立刻 TRANSFORM**

```python
def miner_action(uid, data, state, obs, config):
    col, row, energy = data[1], data[2], data[3]
    pos_key = f"{col},{row}"
    
    # 站在矿点上且能量足够，立刻转化
    if pos_key in obs.miningNodes and energy >= config.transformCost:
        return "TRANSFORM"
    
    # 找最近矿点
    if obs.miningNodes:
        target = min(obs.miningNodes.keys(),
            key=lambda k: manhattan(k, col, row))
        tc, tr = map(int, target.split(','))
        return bfs(state, (col, row), (tc, tr))
    
    # 没找到矿点，北移等待 Scout 发现
    return move_north_safe(col, row, state)
```

**TRANSFER 能量回收机制**（极其重要）

Scout 上限只有 100，满了就浪费。需要把能量传回 Factory（无上限）：

```python
def maybe_transfer(uid, data, state, obs):
    col, row, energy, rtype = data[1], data[2], data[3], data[0]
    
    # Scout 能量达到 80% 上限，寻找相邻友方传输
    if rtype == 1 and energy >= 80:
        for direction, dc, dr in [("NORTH",0,1),("SOUTH",0,-1),("EAST",1,0),("WEST",-1,0)]:
            nc, nr = col+dc, row+dr
            for fuid, fdata in obs.robots.items():
                if fdata[4] == obs.player and fdata[1] == nc and fdata[2] == nr:
                    w = state.known_walls.get((col, row), 0)
                    wall_bits = {"NORTH":1,"EAST":2,"SOUTH":4,"WEST":8}
                    if not (w & wall_bits[direction]):
                        return f"TRANSFER_{direction}"
    return None
```

## 五、滚轴速度预判

这是容易翻车的坑。游戏滚轴速度不是固定的：

```python
def scroll_interval(step, config):
    """返回当前步数对应的滚轴间隔（每 N 步滚动一次）"""
    if step >= config.scrollRampSteps:  # 400步后
        return 1  # 每步都滚！
    # 线性插值
    start = config.scrollStartInterval  # 4
    end = config.scrollEndInterval      # 1
    ratio = step / config.scrollRampSteps
    return max(1, int(start - (start - end) * ratio))

def turns_until_scroll(step, config):
    interval = scroll_interval(step, config)
    return interval - (step % interval)

def danger_level(factory_row, south_bound, step, config):
    """工厂距南边界多少步会危险"""
    rows_above = factory_row - south_bound
    # 预估未来 20 步的滚动次数
    scrolls = sum(1 for s in range(step, step+20) 
                  if s % max(1, scroll_interval(s, config)) == 0)
    return rows_above - scrolls  # 剩余安全边距
```

**关键阈值：** 400步后每步都滚，Factory 必须不断北移，此时不能停下来建造。

## 六、Worker 的高级用法

Worker 是被严重低估的单位。100 能量建/拆一面墙，用好了能改变战局：

**进攻性拆墙（己方通道优化）：**
```python
def worker_action(uid, data, state, obs, config):
    col, row, energy = data[1], data[2], data[3]
    w = state.known_walls.get((col, row), 0)
    
    # 优先北上通路
    if w & 1 and energy >= config.wallRemoveCost:
        return "REMOVE_NORTH"
    
    # 封锁：如果对称镜像位置有敌方，堵住门（对称轴附近的门）
    mirror_col = state.width - 1 - col
    if is_near_door(col, row, state) and enemy_nearby(mirror_col, row, obs):
        return build_to_block(col, row, state, config)
    
    return move_north_safe(col, row, state)
```

**防守性建墙（阻止 Factory Crush）：** 敌方 Factory 无敌，但如果它来压你，Worker 可以在其路径上建墙，迫使它浪费跳跃 CD。

## 七、战斗回避：Crush 机制利用

关键规则：**同类型互相消灭，包括友军！**

```python
def predict_crush_danger(uid, col, row, all_actions, obs):
    """检查下一步移动是否会踩到其他机器人"""
    move_map = {"NORTH": (0,1), "SOUTH": (0,-1), "EAST": (1,0), "WEST": (-1,0)}
    
    next_positions = {}
    for other_uid, action in all_actions.items():
        if other_uid == uid:
            continue
        odata = obs.robots.get(other_uid)
        if not odata:
            continue
        dc, dr = move_map.get(action, (0,0))
        next_pos = (odata[1]+dc, odata[2]+dr)
        next_positions.setdefault(next_pos, []).append(odata[0])  # type
    
    # 检查目标位置是否危险
    # 对于 Scout(1)：任何有其他机器人的格子都危险
    return (col, row) in next_positions
```

**利用 Crush 消灭敌方：** 己方有 Worker/Miner 时，可以走向敌方 Scout（弱于 Worker），消灭它。Factory 可以用来 Crush 敌方任何非 Factory 单位。

## 八、完整 Agent 框架

```python
class CrawlAgent:
    def __init__(self):
        self.state = None
        self.initialized = False
    
    def __call__(self, obs, config):
        if not self.initialized:
            self.state = GameState(config)
            self.initialized = True
        
        self.state.update(obs)
        actions = {}
        
        my_robots = {uid: d for uid, d in obs.robots.items() 
                     if d[4] == obs.player}
        
        # 两阶段决策：先计算意图，再检查冲突
        intended = {}
        for uid, data in my_robots.items():
            rtype = data[0]
            transfer = maybe_transfer(uid, data, self.state, obs)
            if transfer:
                intended[uid] = transfer
            elif rtype == 0:
                intended[uid] = factory_action(uid, data, self.state, obs, config)
            elif rtype == 1:
                intended[uid] = scout_action(uid, data, self.state, obs, config)
            elif rtype == 2:
                intended[uid] = worker_action(uid, data, self.state, obs, config)
            elif rtype == 3:
                intended[uid] = miner_action(uid, data, self.state, obs, config)
        
        # 冲突检测：避免友军互相踩死
        actions = resolve_conflicts(intended, my_robots, self.state, obs)
        return actions

agent_instance = CrawlAgent()

def agent(obs, config):
    return agent_instance(obs, config)
```

## 九、关键注意事项总结

**必须处理的边界 Bug：**

1. **Factory JUMP 落点越界** → 直接 Game Over，每次 JUMP 前必须验证 `row ± 2` 在 `[southBound, northBound]` 内
2. **友军 Crush** → Scout 撞 Scout = 双死，建造时检查北方格子是否已有己方单位
3. **BUILD_SCOUT 时北方有敌方** → 新 Scout 会被秒杀，建造前要确认北格没有敌方
4. **能量耗尽** → Worker 能量低于 100 就无法拆墙，要提前 TRANSFER 补充
5. **400步后的紧急模式** → 必须停止所有非移动行为，全力北移

**高分核心策略优先级：**

```
矿场建立 > Factory 存活 > 能量收集 > 地图控制 > 战斗
```

尽早找到矿点、建立矿场，一个 Mine 每回合 50 能量，存满 1000 后仍持续产出给站在上面的机器人，是整局游戏最强的经济来源，远超打架和捡晶石。
